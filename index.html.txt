        });

        document.getElementById('close-notifications').addEventListener('click', function(e) {
            e.stopPropagation();
            document.getElementById('notifications-sidebar').classList.add('-translate-x-full');
        });

        // Close notifications when clicking outside
        document.addEventListener('click', function(e) {
            const sidebar = document.getElementById('notifications-sidebar');
            const button = document.getElementById('notifications-button');
            
            if (!sidebar.contains(e.target) && e.target !== button && !button.contains(e.target)) {
                sidebar.classList.add('-translate-x-full');
            }
        });

        // Simulate system status change
        setInterval(() => {
            const statusIndicator = document.getElementById('status-indicator');
            const statusText = document.getElementById('system-status');
            const isActive = statusIndicator.classList.contains('status-active');
            
            if (Math.random() > 0.9) { // 10% chance to change status
                if (isActive) {
                    statusIndicator.classList.remove('status-active');
                    statusIndicator.classList.add('status-inactive');
                    statusText.textContent = 'متوقف مؤقتًا';
                } else {
                    statusIndicator.classList.remove('status-inactive');
                    statusIndicator.classList.add('status-active');
                    statusText.textContent = 'يعمل الآن';
                }
            }
        }, 10000);

        // Update live data periodically
        setInterval(() => {
            const inspectedElement = document.querySelector('.text-purple-800');
            const rejectedElement = document.querySelector('.text-red-500');
            const rateElement = document.querySelector('.text-green-600');
            
            // Random small variations
            const inspected = parseInt(inspectedElement.textContent.replace(/,/g, ''));
            const rejected = parseInt(rejectedElement.textContent.replace(/,/g, ''));
            const rate = parseInt(rateElement.textContent.replace(/,/g, ''));
            
            const newInspected = inspected + Math.floor(Math.random() * 20);
            const newRejected = rejected + Math.floor(Math.random() * 3);
            const newRate = rate + Math.floor(Math.random() * 5) - 2;
            
            inspectedElement.textContent = newInspected.toLocaleString();
            rejectedElement.textContent = newRejected.toLocaleString();
            rateElement.textContent = Math.max(90, Math.min(130, newRate)).toLocaleString();
            
            // Update rejection percentage
            const percentage = ((newRejected / newInspected) * 100).toFixed(1);
            document.querySelector('.text-sm.text-gray-500.mt-2').textContent = ${percentage}% معدل الرفض;
            
        }, 5000);

        feather.replace();
